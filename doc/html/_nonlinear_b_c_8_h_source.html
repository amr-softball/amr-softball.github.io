208  );

259  virtual void solve(Real& ghost_enthalpy,
260  const Real a_interior_temperature,
261  const Real a_ghost_bulk_conc)
262  {
263  CH_TIME("NonlinearBCSolverNewton::solve");
264 
265  // This is somewhat arbitrary
266  Real dH = 0.01;
267 
268  Real res_old = m_max_resid*10, res_new = m_max_resid*10;
269  int num_iter = 0;
270 
271  while (abs(res_old) > m_max_resid && num_iter < m_max_iter)
272  {
273 
274  // Need residual at H and H+dH
275  m_bc->computeResidual(res_old,
276  ghost_enthalpy, // enthalpy in ghost cell
277  a_ghost_bulk_conc, // bulk conc in ghost cell
278  a_interior_temperature // temperature at interior grid cell
279  );
280 
281  m_bc->computeResidual(res_new,
282  ghost_enthalpy+dH, // enthalpy in ghost cell
283  a_ghost_bulk_conc, // bulk conc in ghost cell
284  a_interior_temperature // temperature at interior grid cell
285  );
286 
287  // Near the eutectic, can have the same temperature for different enthalpies and therefore
288  // the same residual. Want to avoid dividing by 0, so just do picard iterations in this case.
289  if (abs(res_new - res_old) < m_max_resid)
290  {
291  ghost_enthalpy = ghost_enthalpy + res_old;
292  // Try increasing dH to stop this happening next time
293  dH = 2*dH;
294  }
295  else
296  {
297  ghost_enthalpy = ghost_enthalpy - res_old * (dH)/(res_new - res_old);
298  }
299 
300  num_iter++;
301  }
302 
303  }
